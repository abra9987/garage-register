**Step 1: Modify src/app/globals.css** -- Replace dark mode approach (per D-08, UIUX-02)

After shadcn/ui init, the globals.css will have `.dark { ... }` block. Replace it with `@media (prefers-color-scheme: dark)` per RESEARCH.md Pattern 4.

Keep the `@import "tailwindcss"` and `@theme inline { ... }` sections that shadcn/ui generated. Only change the dark mode block:

- Find the `.dark { ... }` or `.dark, .dark * { ... }` block
- Replace the selector with `@media (prefers-color-scheme: dark) { :root { ... } }`
- Keep all the CSS variable values inside (they define the dark theme colors)
- Make sure the light theme values are under `:root { ... }` (not inside any media query)
- REMOVE any `@custom-variant dark` line if present (this would use the .dark class selector)

The result should be:
```css
@import "tailwindcss";

@theme inline {
  /* ... shadcn/ui theme tokens (keep as generated) ... */
}

:root {
  /* ... light mode CSS variable values (keep as generated) ... */
}

@media (prefers-color-scheme: dark) {
  :root {
    /* ... dark mode CSS variable values (move from .dark block) ... */
  }
}

/* Keep any other base styles shadcn/ui generated */
```

**Step 5: Create src/lib/audit.ts** -- Append-only audit logging utility (per D-10, AUDT-01, AUDT-02, AUDT-03)

```typescript
import { db } from "@/lib/db";
import { auditLog } from "@/lib/db/schema";

type AuditAction = "created" | "updated" | "deleted" | "exported";

interface AuditEntry {
  entityType: string;
  entityId: string;
  action: AuditAction;
  userId: string;
  fieldName?: string;
  oldValue?: string;
  newValue?: string;
}

/**
 * Log an audit entry. Append-only -- this function only inserts, never updates or deletes.
 * AUDT-03: Immutability enforced at application layer.
 */
export async function logAudit(entry: AuditEntry): Promise<void> {
  await db.insert(auditLog).values({
    entityType: entry.entityType,
    entityId: entry.entityId,
    action: entry.action,
    userId: entry.userId,
    fieldName: entry.fieldName ?? null,
    oldValue: entry.oldValue ?? null,
    newValue: entry.newValue ?? null,
  });
}

/**
 * Log multiple field changes for the same entity in a single batch.
 * Used when multiple fields are edited at once during review.
 */
export async function logAuditBatch(
  entries: AuditEntry[]
): Promise<void> {
  if (entries.length === 0) return;
  await db.insert(auditLog).values(
    entries.map((entry) => ({
      entityType: entry.entityType,
      entityId: entry.entityId,
      action: entry.action,
      userId: entry.userId,
      fieldName: entry.fieldName ?? null,
      oldValue: entry.oldValue ?? null,
      newValue: entry.newValue ?? null,
    }))
  );
}
```
