- `dateTo` (string): "YYYY-MM-DD" or undefined

Build where conditions:
```typescript
const conditions: SQL[] = [];

if (q) {
  conditions.push(
    or(
      ilike(vehicles.vin, `%${q}%`),
      ilike(vehicles.jobNumber, `%${q}%`),
      ilike(vehicles.make, `%${q}%`),
      ilike(vehicles.model, `%${q}%`),
      ilike(vehicles.sellerName, `%${q}%`),
      ilike(vehicles.buyerName, `%${q}%`),
    )!
  );
}

if (status) {
  conditions.push(eq(vehicles.status, status));
}

if (dateFrom) {
  conditions.push(gte(vehicles.createdAt, new Date(dateFrom + "T00:00:00")));
}
if (dateTo) {
  conditions.push(lte(vehicles.createdAt, new Date(dateTo + "T23:59:59")));
}

const where = conditions.length > 0 ? and(...conditions) : undefined;
```

Build sort column mapping (map string name to Drizzle column reference):
```typescript
const sortColumns: Record<string, typeof vehicles.createdAt> = {
  jobNumber: vehicles.jobNumber,
  vin: vehicles.vin,
  year: vehicles.year,
  make: vehicles.make,
  model: vehicles.model,
  status: vehicles.status,
  purchaseDate: vehicles.purchaseDate,
  saleDate: vehicles.saleDate,
  createdAt: vehicles.createdAt,
};
const sortCol = sortColumns[sort] ?? vehicles.createdAt;
const orderByClause = dir === "asc" ? asc(sortCol) : desc(sortCol);
```

Execute paginated query (25 per page per D-49):
```typescript
const pageSize = 25;
const offset = (page - 1) * pageSize;

const [results, [{ total }]] = await Promise.all([

    id: vehicles.id,
    jobNumber: vehicles.jobNumber,
    status: vehicles.status,
    vin: vehicles.vin,
    year: vehicles.year,
    make: vehicles.make,
    model: vehicles.model,
    purchaseDate: vehicles.purchaseDate,
    saleDate: vehicles.saleDate,
    createdAt: vehicles.createdAt,
    updatedAt: vehicles.updatedAt,
  })
    .from(vehicles)
    .where(where)
    .orderBy(orderByClause)
    .limit(pageSize)
    .offset(offset),
  db.select({ total: count() }).from(vehicles).where(where),
]);
```

Return: `apiSuccess({ vehicles: results, total: Number(total), page, pageSize })`.
